Create `.github/workflows/docker-build.yml`:

```yaml
name: Build and Push Docker Image

on:
  push:
    branches: [main]
    tags: ['v*']
  pull_request:
    branches: [main]

concurrency:
  group: docker-build-${{ github.ref }}
  cancel-in-progress: true

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up QEMU
        uses: docker/setup-qemu-action@v3

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Login to GitHub Container Registry
        if: github.event_name != 'pull_request'
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata for Docker
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ghcr.io/stragix-innovations/bifrost
          tags: |
            type=ref,event=branch
            type=semver,pattern=v{{version}}
            type=semver,pattern=v{{major}}.{{minor}}
            type=sha,prefix=
            type=raw,value=latest,enable=${{ github.ref == format('refs/heads/{0}', 'main') }}

      - name: Build and push Docker image
        uses: docker/build-push-action@v6
        with:
          context: .
          file: transports/Dockerfile
          platforms: linux/amd64,linux/arm64
          push: ${{ github.event_name != 'pull_request' }}
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          build-args: |
            VERSION=${{ github.ref_name }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
```

Key design choices:
- Uses `docker/metadata-action` for smart tag generation (per D-17): semver tags from git tags, SHA for traceability, `latest` only from main
- QEMU emulation for arm64 cross-compilation (Pitfall 5 from research: CGO + musl works under QEMU)
- GHA cache for faster rebuilds
- PR builds validate the Dockerfile but don't push (per D-16)
- `context: .` with `file: transports/Dockerfile` matches the existing Dockerfile's COPY paths (it copies from repo root)

Per D-14, create a weekly CI job that detects upstream drift.

Create `.github/workflows/upstream-check.yml`:

```yaml
name: Upstream Drift Check

on:
  schedule:
    - cron: '0 9 * * 1'  # Weekly on Monday at 9am UTC
  workflow_dispatch:       # Manual trigger

jobs:
  check-upstream:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      issues: write
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Fetch upstream
        run: |
          git remote add upstream https://github.com/maximhq/bifrost.git || true
          git fetch upstream main

      - name: Check for upstream changes
        id: check
        run: |
          # Count commits ahead/behind
          BEHIND=$(git rev-list --count HEAD..upstream/main)
          AHEAD=$(git rev-list --count upstream/main..HEAD)
          echo "behind=$BEHIND" >> $GITHUB_OUTPUT
          echo "ahead=$AHEAD" >> $GITHUB_OUTPUT

          if [ "$BEHIND" -eq 0 ]; then
            echo "status=up-to-date" >> $GITHUB_OUTPUT
            echo "Up to date with upstream"
            exit 0
          fi

          echo "Behind upstream by $BEHIND commits, ahead by $AHEAD commits"

      - name: Test merge
        if: steps.check.outputs.behind != '0'
        id: merge
        run: |
          git config user.email "ci@stragix.com"
          git config user.name "CI Bot"
          if git merge upstream/main --no-commit --no-ff 2>/dev/null; then
            echo "result=clean" >> $GITHUB_OUTPUT
            echo "Clean merge possible"
            git merge --abort
          else
            echo "result=conflicts" >> $GITHUB_OUTPUT
            echo "Merge has conflicts"
            # List conflicting files
            git diff --name-only --diff-filter=U > /tmp/conflicts.txt 2>/dev/null || true
            git merge --abort
            if [ -f /tmp/conflicts.txt ]; then
              echo "Conflicting files:"
              cat /tmp/conflicts.txt
              echo "conflict_files<<EOF" >> $GITHUB_OUTPUT
              cat /tmp/conflicts.txt >> $GITHUB_OUTPUT
              echo "EOF" >> $GITHUB_OUTPUT
            fi
          fi

      - name: Report status
        if: steps.check.outputs.behind != '0'
        run: |
          BEHIND=${{ steps.check.outputs.behind }}
          RESULT=${{ steps.merge.outputs.result }}
          if [ "$RESULT" = "conflicts" ]; then
            echo "::warning::Upstream merge has conflicts ($BEHIND commits behind). Conflicting files: ${{ steps.merge.outputs.conflict_files }}"
          else
            echo "::notice::Upstream has $BEHIND new commits. Clean merge possible."
          fi
```

Key design choices:
- Weekly schedule matches D-14 (manual periodic merge)
- `workflow_dispatch` allows on-demand checks before planned merges
- Tests merge without committing to detect conflicts early
- Reports conflicting file list for quick triage
- Uses GitHub annotations (::warning::, ::notice::) for visibility in Actions UI
